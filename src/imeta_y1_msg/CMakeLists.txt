cmake_minimum_required(VERSION 3.0.2)
project(imeta_y1_msg)

# set for clangd
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

  message_generation
  std_msgs
)

  ArmControl.msg
  ArmState.msg
)

  DEPENDENCIES
  std_msgs
)
